CREATE TABLE nb_products (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,          -- PK
  product_category_id INT UNSIGNED NOT NULL,           -- 제품 카테고리 ID
  title VARCHAR(100) NOT NULL,                         -- 제품명
  sub_title VARCHAR(100) DEFAULT NULL,                 -- 서브 타이틀
  description TEXT DEFAULT NULL,                       -- 설명
  url VARCHAR(200) DEFAULT NULL,                       -- URL
  feature TEXT DEFAULT NULL,                           -- 특징
  feature_desc TEXT DEFAULT NULL,                      -- 특징 설명

  -- 이미지들 (추가됨)
  thumb_image   VARCHAR(255) DEFAULT NULL,             -- 썸네일 이미지
  detail_image  VARCHAR(255) DEFAULT NULL,             -- 상세 이미지
  banner_img    VARCHAR(255) DEFAULT NULL,             -- 배너 이미지
  product_img_1 VARCHAR(255) DEFAULT NULL,
  product_img_2 VARCHAR(255) DEFAULT NULL,
  product_img_3 VARCHAR(255) DEFAULT NULL,
  product_img_4 VARCHAR(255) DEFAULT NULL,
  product_img_5 VARCHAR(255) DEFAULT NULL,

  -- JSON 필드들
  selected_variants JSON DEFAULT NULL,                 -- 세컨더리 옵션 (체크박스 선택값)
  variant_specs JSON DEFAULT NULL,                     -- 기술 정보 (타이틀/설명 쌍)

  -- 관리 필드
  sort_no INT NOT NULL DEFAULT 0,                      -- 정렬 번호
  is_active TINYINT(1) NOT NULL DEFAULT 1,             -- 노출 여부
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
             ON UPDATE CURRENT_TIMESTAMP,

  KEY idx_category (product_category_id),
  KEY idx_sort (sort_no)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
